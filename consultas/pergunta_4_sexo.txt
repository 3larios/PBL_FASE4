SELECT 
    p.FL_SEXO,
    c.DS_CATEGORIA,
    pt.NM_PONTO_TURISTICO,
    pt.NR_LATITUDE,
    pt.NR_LONGITUDE,
    a.NR_NOTA_AVALIACAO
FROM 
    ALFA_AVALIACAO a
JOIN 
    ALFA_PESSOA p ON a.NM_LOGIN = p.NM_LOGIN
JOIN 
    ALFA_EVENTO_PONTO_TURISTICO ept ON a.ID_EVENTO_PONTO_TURISTICO = ept.ID_EVENTO_PONTO_TURISTICO
JOIN 
    ALFA_PONTO_TURISTICO pt ON ept.ID_PONTO_TURISTICO = pt.ID_PONTO_TURISTICO
JOIN 
    ALFA_CATEGORIA c ON pt.ID_CATEGORIA = c.ID_CATEGORIA
WHERE 
    a.NR_NOTA_AVALIACAO >= 0 
    AND a.NR_NOTA_AVALIACAO IS NOT NULL
    AND pt.ST_PONTO_TURISTICO = 'A' -- Filtra apenas pontos turísticos ativos
    AND ept.ST_EVENTO = 'A';        -- Filtra apenas eventos ativos
